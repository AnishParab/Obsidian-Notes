# Code
``` python
import numpy as np

my_list = [1, 2, 3, 4]
print(type(my_list))

my_list = my_list * 2
print(my_list)

print()

# NumPy Array
array = np.array([1, 2, 3, 4])
print(type(array))

array = array * 2
print(array)
```

**Output**:
``` text
<class 'list'>
[1, 2, 3, 4, 1, 2, 3, 4]

<class 'numpy.ndarray'>
[2 4 6 8]
```
