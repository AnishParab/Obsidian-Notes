# Bar Chart
``` python
import matplotlib.pyplot as plt
import seaborn as sb
import numpy as np

x = ['A', 'B', 'C', 'D']
y = [1, 5, 2, 8]

# Bar Chart
plt.bar(x,y)

plt.xlabel("Categories")
plt.ylabel("Values")
plt.title("Bar Plot")
plt.show()
```
